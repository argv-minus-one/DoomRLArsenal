ACTOR RLSuperchargedNuclearPlasmaPistolPickup : CustomInventory
{
	Radius 16
	Height 8
	Inventory.PickupMessage ""
	Inventory.PickupSound ""
	States
	{
	Spawn:
		NUPP X 5
		Loop
	Pickup:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("RLSuperchargedNuclearPlasmaPistol",1,"NotPickup") // If the player already has this weapon, they are not given the weapon
		TNT1 A 0 A_JumpIfInventory("RLWeaponLimit",0,"NotPickup") // The absolute max limit check for weapons
		TNT1 A 0 A_PlaySound("weapons/superchargednukeplasmapistolpickup")
		TNT1 A 0 A_GiveInventory("RLSuperchargedNuclearPlasmaPistol",1)
		TNT1 A 0 A_GiveInventory("RLWeaponLimit",1)
		TNT1 A 0 A_GiveInventory("RLNuclearWeaponSetBonusWeapon",1)
		TNT1 A 0 A_JumpIfInventory("RLNuclearWeaponSetBonusArmor",1,2)
		TNT1 A 0 A_Print("Picked up a \cvSupercharged Nuclear Plasma Pistol.")
		Stop
		TNT1 A 0 A_GiveInventory("RLNuclearWeaponSetBonusActive",1)
		TNT1 A 0 A_Print("Picked up a \cvSupercharged Nuclear Plasma Pistol.\n\c-= FURIOUS FUSION =\n\cgNuclear weaponry recharges twice as fast", 8, "CONFONT")
		Stop
	NotPickup:
		TNT1 A 0 A_JumpIfInventory("DRLA_WeaponAntiLogFloodCheck",1,"NotPickup_NoPrint")
		TNT1 A 0 A_GiveInventory("DRLA_WeaponAntiLogFloodCheck")
		TNT1 A 0 A_Print("\cv= Supercharged Nuclear Plasma Pistol =", 2, "CONFONT")
	 NotPickup_NoPrint:	
		TNT1 A 0
		Fail
	}
}
