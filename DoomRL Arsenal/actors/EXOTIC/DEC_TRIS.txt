ACTOR RLTristarBlasterPickup : CustomInventory
{
	Radius 16
	Height 8
	Inventory.PickupMessage ""
	Inventory.PickupSound ""
	States
	{
	Spawn:
		TRIS X -1
		Stop
	Pickup:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("RLTristarBlaster",1,"NotPickup") // If the player already has this weapon, they are not given the weapon
		TNT1 A 0 A_JumpIfInventory("RLWeaponLimit",0,"NotPickup2") // The absolute max limit check for weapons
		TNT1 A 0 A_PlaySound("weapons/tristarblasterpickup")
		TNT1 A 0 A_GiveInventory("RLTristarBlaster",1)
		TNT1 A 0 A_GiveInventory("RLWeaponLimit",1)
		TNT1 A 0 A_GiveInventory("RLCerberusSetBonusTristarBlaster",1)
		TNT1 A 0 A_JumpIfInventory("RLCerberusSetBonusBoots",1,"SetBonusCheck")
		TNT1 A 0 A_JumpIfInventory("RLCerberusSetBonusArmor",1,"SetBonusCheck")
	PickupFinish:
		TNT1 A 0 A_Print("Picked up a \ctTristar Blaster.")
		Stop
	NotPickup:
		TNT1 A 0 A_JumpIfInventory("Cell",0,1)
		Goto AmmoPickup
	NotPickup2:
		TNT1 A 0 A_JumpIfInventory("DRLA_WeaponAntiLogFloodCheck",1,"NotPickup_NoPrint")
		TNT1 A 0 A_GiveInventory("DRLA_WeaponAntiLogFloodCheck")
		TNT1 A 0 A_Print("\ct= Tristar Blaster =", 2, "CONFONT")
	 NotPickup_NoPrint:	
		TNT1 A 0
		Fail
	AmmoPickup:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("Cell",0,"AmmoFull")
		TNT1 A 0 A_PlaySound("misc/w_pkup")
		TNT1 A 1 A_GiveInventory("Cell",45)
		Stop
	AmmoFull:
		TRIS X 1
		Fail
	SetBonusCheck:
		TNT1 A 0 A_JumpIfInventory("RLCerberusSetBonusBoots",1,1)
		Goto PickupFinish
		TNT1 A 0 A_JumpIfInventory("RLCerberusSetBonusArmor",1,1)
		Goto PickupFinish
		TNT1 A 0 A_GiveInventory("RLCerberusSetBonusActive",1)
		TNT1 A 0 A_Print("Picked up a \ctTristar Blaster.\n\c-= TRINITY GUARDIAN =\n\cgTristar Blaster fires three times faster", 8, "CONFONT")
		TNT1 A 0
		Stop
	}
}
