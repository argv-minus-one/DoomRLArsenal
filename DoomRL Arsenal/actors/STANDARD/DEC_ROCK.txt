ACTOR RLRocketLauncherPickup : CustomInventory
{
	Radius 16
	Height 8
	Inventory.PickupMessage ""
	Inventory.PickupSound ""
	States
	{
	Spawn:
		MISG X -1
		Stop
	Pickup:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("RLRocketLauncher",1,"NotPickup") // If the player already has this weapon, they are not given the weapon
		TNT1 A 0 A_JumpIfInventory("RLWeaponLimit",0,"NotPickup3") // The absolute max limit check for weapons
		TNT1 A 0 A_PlaySound("weapons/rocketlauncherpickup")
		TNT1 A 0 A_GiveInventory("RLRocketLauncher",1)
		TNT1 A 0 A_GiveInventory("RLWeaponLimit",1)
		TNT1 A 0 A_Print("Picked up a \cjRocket Launcher.")
		Stop
	NotPickup:
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",0,1)
		Goto AmmoPickup
	NotPickup2:
		TNT1 A 0 A_JumpIfInventory("DRLA_WeaponAntiLogFloodCheck",1,"NotPickup_NoPrint")
		TNT1 A 0 A_GiveInventory("DRLA_WeaponAntiLogFloodCheck")
		TNT1 A 0 A_Print("\cj= Rocket Launcher =", 2, "CONFONT")
	 NotPickup_NoPrint:	
		TNT1 A 0
		Fail
	NotPickup3:
		TNT1 A 0 A_JumpIfInventory("RLMissileLauncher",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLNapalmLauncher",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLUnknownHerald",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLRevenantsLauncher",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLHurricaneCannon",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLMicroLauncher",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLTacticalRocketLauncher",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLMiniMissilePistol",1,"NotPickup")
		TNT1 A 0 A_JumpIfInventory("RLGrenadeLauncher",1,"NotPickup")
		Goto NotPickup2
	AmmoPickup:
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",0,"AmmoFull")
		TNT1 A 0 A_PlaySound("misc/w_pkup")
		TNT1 A 1 A_GiveInventory("RocketAmmo",5)
		Stop
	AmmoFull:
		MISG X 1
		Fail
	}
}
